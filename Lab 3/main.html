Autocorrelation Function

function R = autocorrelation(sample)
N = length(sample);
R = zeros(length(N), 1);
sample_avg = average(sample);
for n = 0:round(sqrt(N))
sum1 = 0;
sum2 = 0;
for k = 1:(N-n)
sum1 = sum1 + (sample(k)-sample_avg)*(sample(k+n)-sample_avg);
sum2 = sum2 + (sample(k)-sample_avg)^2;
end
R(n+1) = sum1/sum2;
end
R = R';
end
